ADD_DEFINITIONS(-DSTANDALONE -DCOVISE_DEVSERVER -D_OLD_TERMIOS)

IF(WIN32)
  ADD_DEFINITIONS(-DUTIL_EXPORTS -DCOVISE_UTIL)
  SET(EXTRA_LIBS ws2_32)
ENDIF(WIN32)

SET(FOBSERVER_SOURCES
  FOBserver.cpp
  fob.cpp
)

SET(FOBSERVER_HEADERS
  fob.h
)

IF(WIN32)
  SET(FOBSERVER_SOURCES ${FOBSERVER_SOURCES} fobWindows.cpp)
ELSE(WIN32)
  SET(FOBSERVER_SOURCES ${FOBSERVER_SOURCES} fobUnix.cpp)
ENDIF(WIN32)

ADD_COVISE_EXECUTABLE(FOBserver ${FOBSERVER_SOURCES} ${FOBSERVER_HEADERS})
TARGET_LINK_LIBRARIES(FOBserver coDevServer ${EXTRA_LIBS}) 

COVISE_INSTALL_TARGET(FOBserver)
