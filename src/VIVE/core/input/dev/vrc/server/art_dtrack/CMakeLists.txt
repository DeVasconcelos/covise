ADD_DEFINITIONS(-DSTANDALONE -DCOVISE_DEVSERVER)

IF(WIN32)
  ADD_DEFINITIONS(-DUTIL_EXPORTS -DCOVISE_UTIL)
  SET(EXTRA_LIBS ws2_32)
ENDIF(WIN32)

SET(ART_DTRACKSERVER_SOURCES
  ART_DTRACKserver.cpp
  DTrack.cpp
)

SET(ART_DTRACKSERVER_HEADERS
  DTrack.hpp
)

ADD_COVISE_EXECUTABLE(ART_DTRACKserver ${ART_DTRACKSERVER_SOURCES} ${ART_DTRACKSERVER_HEADERS})
TARGET_LINK_LIBRARIES(ART_DTRACKserver coDevServer ${EXTRA_LIBS}) 

COVISE_INSTALL_TARGET(ART_DTRACKserver)
